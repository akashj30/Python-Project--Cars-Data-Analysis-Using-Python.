Treating the Missing Value¶

In [8]:
# categorical= mode()
# continuous= mean()

for i in df:
    if df[i].dtypes == 'object':
        m=df[i].mode()[0]
        df[i]=df[i].fillna(m)
    else:
        me=df[i].mean()
        df[i]=df[i].fillna(me)
